const mongoose = require('mongoose');

// Define a Mongoose schema for the combined data structure
const combinedSchema = new mongoose.Schema({
  Categories: String,
  SubCategories: String,
  Description: String,
  Title: String,
  Price: Number,
  Images: [String],
  Quantity: Number,
  timestamp: {
    type: Date,
    default: Date.now,
  },
  firstName: String,
  key: String,
  label: String,
  phoneNumber: String,
  state: String,
});

// Create a Mongoose model using the schema
const CombinedModel = mongoose.model('Combined', combinedSchema);
